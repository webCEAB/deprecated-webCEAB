Abrir consola git Bash
$ cd
$ cd webCEAB/
	OJO CHECAR QUE ESTOY EN LA RAMA Vistas
$ git init
$ git status #Nos mostrará los archivos que han tenido cambios (en rojo)
$ git add -A #reconoce incluso archivos eliminados
$ git status #nos muestra los archivos que están listos (en verde)
$ git commit -m "Aquí el comentario de los cambios hechos" #Envía los archivos, hace un comentario sobre ellos, genera un punto en nuestra línea de tiempo
$ git push origin Vistas #prepara los cambios para que sean "jalados" por el código a pythonanywhere
